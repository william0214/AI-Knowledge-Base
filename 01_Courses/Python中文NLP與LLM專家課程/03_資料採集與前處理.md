```python
# 來源：
# 用途：中文文本清洗 pipeline

import re

def clean_text(text: str) -> str:
    """中文文本清洗"""
    # 去除 HTML 標籤
    text = re.sub(r'<[^>]+>', '', text)
    # 去除特殊字元（保留中文、英文、數字）
    text = re.sub(r'[^\w\s\u4e00-\u9fff]', '', text)
    # 去除多餘空白
    text = re.sub(r'\s+', ' ', text).strip()
    return text
```
